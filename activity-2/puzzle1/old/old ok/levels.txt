        // دالة الانتقال إلى مستوى معين
        function goToLevel(levelNumber) {
            if (levelNumber === 1) {
                window.location.href = "level1.html"; // الانتقال للمستوى الأول
            } else if (levelNumber === 2) {
                window.location.href = "level2.html"; // الانتقال للمستوى الثاني
            } else if (levelNumber === 3) {
                window.location.href = "level3.html"; // الانتقال للمستوى الثالث
            }
        }

        // دالة الخروج من اللعبة
        function exitGame() {
            // في المتصفح لا يمكن إغلاق النافذة إلا إذا فُتحت بواسطة JavaScript
            // ممكن نعرض رسالة تأكيد
            if (confirm("هل تريد الخروج من اللعبة؟")) {
                window.close(); // محاولة إغلاق النافذة
